            'orderby' => 'meta_value_num',
            'order' => 'DESC'
        ));
        
        if ($popular_posts) :
        ?>
            <ul class="popular-posts-list">
                <?php foreach ($popular_posts as $post) : setup_postdata($post); ?>
                    <li class="popular-post-item">
                        <?php if (has_post_thumbnail()) : ?>
                            <div class="popular-post-thumb">
                                <a href="<?php the_permalink(); ?>">

                                </a>
                            </div>
                        <?php endif; ?>
                        <div class="popular-post-content">
                            <h4 class="popular-post-title">
                                <a href="<?php the_permalink(); ?>"><?php the_title(); ?></a>
                            </h4>
                            <div class="popular-post-meta">
                                <span class="post-date"><?php echo get_the_date('j M Y'); ?></span>
                                <span class="post-views"><?php echo tema_get_post_views(get_the_ID()); ?></span>
                            </div>
                        </div>
                    </li>
                <?php endforeach; wp_reset_postdata(); ?>
            </ul>
        <?php endif; ?>
    </div>
    
    <!-- Posts recentes -->
    <div class="widget recent-posts-widget">
        <h3 class="widget-title">Posts Recentes</h3>
        <?php
        $recent_posts = wp_get_recent_posts(array(
            'numberposts' => 5,
            'post_status' => 'publish'
        ));
        
        if ($recent_posts) :
        ?>
            <ul class="recent-posts-list">
                <?php foreach ($recent_posts as $post) : ?>
                    <li class="recent-post-item">
                        <?php if (has_post_thumbnail($post['ID'])) : ?>
                            <div class="recent-post-thumb">
                                <a href="<?php echo get_permalink($post['ID']); ?>">
                                    <?php echo get_the_post_thumbnail($post['ID'], 'thumbnail'); ?>
                                </a>
                            </div>
                        <?php endif; ?>
                        <div class="recent-post-content">
                            <h4 class="recent-post-title">
                                <a href="<?php echo get_permalink($post['ID']); ?>">
                                    <?php echo $post['post_title']; ?>
                                </a>
                            </h4>
                            <div class="recent-post-meta">
                                <span class="post-date"><?php echo get_the_date('j M Y', $post['ID']); ?></span>
                            </div>
                        </div>
                    </li>
                <?php endforeach; ?>
            </ul>
        <?php endif; ?>
    </div>
    
    <!-- Widget de categorias -->
    <div class="widget categories-widget">
        <h3 class="widget-title">Categorias</h3>
        <ul class="categories-list">
            <?php
            $categories = get_categories(array(
                'orderby' => 'name',
                'order'   => 'ASC',
                'hide_empty' => true,
            ));
            
            foreach ($categories as $category) :
            ?>
                <li class="category-item">
                    <a href="<?php echo get_category_link($category->term_id); ?>" class="category-link">
                        <span class="category-name"><?php echo $category->name; ?></span>
                        <span class="category-count">(<?php echo $category->count; ?>)</span>
                    </a>
                </li>
            <?php endforeach; ?>
        </ul>
    </div>
    
    <!-- Widget de tags -->
    <div class="widget tags-widget">
        <h3 class="widget-title">Tags</h3>
        <div class="tags-cloud">
            <?php
            $tags = get_tags(array(
                'orderby' => 'name',
                'order' => 'ASC',
                'number' => 20,
            ));
            
            foreach ($tags as $tag) :
            ?>
                <a href="<?php echo get_tag_link($tag->term_id); ?>" class="tag-link">
                    <?php echo $tag->name; ?>
                </a>
            <?php endforeach; ?>
        </div>
    </div>
    
    <!-- Newsletter signup (exemplo) -->
    <div class="widget newsletter-widget">
        <h3 class="widget-title">Newsletter</h3>
        <p class="newsletter-description">
            Receba as últimas novidades e posts diretamente no seu email.
        </p>
        <form class="newsletter-form" action="#" method="post">
            <div class="form-group">
                <input type="email" name="email" placeholder="Seu email" class="form-input" required>
            </div>
            <button type="submit" class="btn btn-primary btn-block">
                Inscrever-se
            </button>
        </form>
    </div>
    
    <!-- Widgets dinâmicos -->
    <?php dynamic_sidebar('sidebar-1'); ?>
    
</aside><!-- #secondary -->
